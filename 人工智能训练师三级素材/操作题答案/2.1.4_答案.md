# 2.1.4 医疗研究数据清洗和标注设计

## 题目要求

1. 加载数据集，查看数据类型、表结构和每列空缺值数量
2. 将"就诊日期"和"诊断日期"规范为"yyyy-mm-dd"格式，将"病人ID"列名改为"患者ID"，显示修改后的表结构
3. 增加"诊断延迟"（诊断日期-就诊日期）和"病程"（当前日期-诊断日期）两列，删除不合理数据（如负数、年龄几百岁等）
4. 检查重复值并删除，记录删除的行数
5. 对 [年龄, 体重, 身高] 进行归一化处理
6. 统计不同疾病类型的治疗结果分布，绘制柱状图
7. 分析年龄和疾病严重程度的关系，绘制散点图
8. 保存处理后的数据为 2.1.4_cleaned_data.csv
9. 制定数据清洗和数据标注规范（写入2.1.4.docx）
10. 将代码和运行结果保存为 2.1.4.html

## 答案代码

```python
# 导入必要的库
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler
from datetime import datetime

# 设置中文字体显示
plt.rcParams['font.sans-serif'] = ['SimHei', 'Microsoft YaHei']
plt.rcParams['axes.unicode_minus'] = False

# ========================
# 第1步：加载数据集，查看数据类型、表结构和空缺值
# ========================
df = pd.read_csv('medical_data.csv', encoding='gbk')
print("===== 数据集前五行 =====")
print(df.head())
print(f"\n数据集形状: {df.shape}")
print("\n===== 数据类型 =====")
print(df.dtypes)
print("\n===== 表结构信息 =====")
df.info()
print("\n===== 每列空缺值数量 =====")
print(df.isnull().sum())

# ========================
# 第2步：日期格式规范化 + 列名修改
# ========================
# 将"就诊日期"和"诊断日期"转换为标准日期格式
df['就诊日期'] = pd.to_datetime(df['就诊日期'], errors='coerce')
df['诊断日期'] = pd.to_datetime(df['诊断日期'], errors='coerce')

# 格式化为 yyyy-mm-dd 字符串（保留datetime类型便于后续计算）
print("\n就诊日期格式化示例:")
print(df['就诊日期'].head())
print("\n诊断日期格式化示例:")
print(df['诊断日期'].head())

# 将"病人ID"列名改为"患者ID"
df = df.rename(columns={'病人ID': '患者ID'})
print("\n===== 修改列名后的表结构 =====")
print(df.columns.tolist())
df.info()

# ========================
# 第3步：增加"诊断延迟"和"病程"列，删除不合理数据
# ========================
# 诊断延迟 = 诊断日期 - 就诊日期（天数）
df['诊断延迟'] = (df['诊断日期'] - df['就诊日期']).dt.days

# 病程 = 当前日期 - 诊断日期（天数）
current_date = pd.Timestamp.now()
df['病程'] = (current_date - df['诊断日期']).dt.days

print("\n===== 新增列前几行 =====")
print(df[['就诊日期', '诊断日期', '诊断延迟', '病程']].head())

# 删除不合理数据
print(f"\n删除不合理数据前的行数: {len(df)}")

# 1. 删除诊断延迟为负数的行（诊断日期早于就诊日期是不合理的，但在某些情况下可能是复诊，这里保留）
# 根据题目要求删除负数
df = df[df['诊断延迟'].isna() | (df['诊断延迟'] >= 0)]

# 2. 删除病程为负数的行
df = df[df['病程'].isna() | (df['病程'] >= 0)]

# 3. 删除年龄不合理的数据（如年龄为几百岁）
# 合理年龄范围：0-120岁
df = df[(df['年龄'] >= 0) & (df['年龄'] <= 120)]

# 4. 删除体重和身高不合理的数据
df = df[df['体重'].isna() | ((df['体重'] > 0) & (df['体重'] < 300))]
df = df[df['身高'].isna() | ((df['身高'] > 0) & (df['身高'] < 250))]

print(f"删除不合理数据后的行数: {len(df)}")

# ========================
# 第4步：检查重复值并删除
# ========================
duplicate_count = df.duplicated().sum()
print(f"\n重复行数量: {duplicate_count}")
df = df.drop_duplicates()
print(f"删除重复值后的数据行数: {len(df)}")
print(f"删除的重复行数: {duplicate_count}")

# ========================
# 第5步：对 [年龄, 体重, 身高] 进行归一化处理
# ========================
cols_to_normalize = ['年龄', '体重', '身高']

# 先处理缺失值（用中位数填充）
for col in cols_to_normalize:
    df[col] = df[col].fillna(df[col].median())

scaler = MinMaxScaler()
df[cols_to_normalize] = scaler.fit_transform(df[cols_to_normalize])
print("\n===== 归一化后的统计信息 =====")
print(df[cols_to_normalize].describe())

# ========================
# 第6步：统计不同疾病类型的治疗结果分布，绘制柱状图
# ========================
# 交叉统计
cross_tab = pd.crosstab(df['疾病类型'], df['治疗结果'])
print("\n===== 不同疾病类型的治疗结果分布 =====")
print(cross_tab)

# 绘制柱状图
fig, ax = plt.subplots(figsize=(12, 6))
cross_tab.plot(kind='bar', ax=ax)
ax.set_title('不同疾病类型的治疗结果分布', fontsize=14)
ax.set_xlabel('疾病类型', fontsize=12)
ax.set_ylabel('数量', fontsize=12)
ax.legend(title='治疗结果')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.savefig('disease_treatment_distribution.png', dpi=100, bbox_inches='tight')
plt.show()
print("柱状图已保存")

# ========================
# 第7步：分析年龄和疾病严重程度的关系，绘制散点图
# ========================
# 将疾病严重程度转换为数值编码
severity_mapping = {'不严重': 1, '一般': 2, '严重': 3, '非常严重': 4}
df['疾病严重程度_数值'] = df['疾病严重程度'].map(severity_mapping)

# 对于未映射到的值，尝试填充
if df['疾病严重程度_数值'].isnull().sum() > 0:
    # 查看未映射的值
    unmapped = df[df['疾病严重程度_数值'].isnull()]['疾病严重程度'].unique()
    print(f"未映射的疾病严重程度值: {unmapped}")
    # 删除无法映射的行
    df = df.dropna(subset=['疾病严重程度_数值'])

fig, ax = plt.subplots(figsize=(10, 6))
ax.scatter(df['年龄'], df['疾病严重程度_数值'], alpha=0.3, s=10)
ax.set_title('年龄与疾病严重程度的关系', fontsize=14)
ax.set_xlabel('年龄（归一化后）', fontsize=12)
ax.set_ylabel('疾病严重程度', fontsize=12)
ax.set_yticks([1, 2, 3, 4])
ax.set_yticklabels(['不严重', '一般', '严重', '非常严重'])
plt.tight_layout()
plt.savefig('age_severity_scatter.png', dpi=100, bbox_inches='tight')
plt.show()
print("散点图已保存")

# ========================
# 第8步：保存处理后的数据
# ========================
# 将日期列转换为字符串格式保存
df['就诊日期'] = df['就诊日期'].dt.strftime('%Y-%m-%d')
df['诊断日期'] = df['诊断日期'].dt.strftime('%Y-%m-%d')

# 删除辅助列
if '疾病严重程度_数值' in df.columns:
    df = df.drop(columns=['疾病严重程度_数值'])

df.to_csv('2.1.4_cleaned_data.csv', index=False, encoding='utf-8-sig')
print("\n数据已保存为 2.1.4_cleaned_data.csv")
print(f"最终数据形状: {df.shape}")
```

## 代码解析

1. **加载数据集**：使用 `pd.read_csv(encoding='gbk')` 读取中文编码的medical_data.csv，包含5440行22列，涵盖患者ID、年龄、性别、疾病类型、治疗方案等医疗字段。通过 `dtypes`、`info()` 和 `isnull().sum()` 查看数据结构。

2. **日期格式规范化**：
   - 使用 `pd.to_datetime(errors='coerce')` 将就诊日期和诊断日期转换为标准datetime格式（yyyy-mm-dd）
   - 使用 `rename()` 将"病人ID"改为"患者ID"

3. **新增列与异常数据清理**：
   - `诊断延迟` = 诊断日期 - 就诊日期（天数差值）
   - `病程` = 当前日期 - 诊断日期（天数差值）
   - 删除诊断延迟为负数的行、病程为负数的行
   - 删除年龄超出0-120岁合理范围的行
   - 删除体重和身高不合理的数据

4. **重复值处理**：检测并删除重复行，记录删除数量。

5. **归一化处理**：使用 `MinMaxScaler` 对年龄、体重、身高三列进行Min-Max归一化，缩放到[0,1]区间。缺失值先用中位数填充。

6. **疾病类型与治疗结果分布**：使用 `pd.crosstab()` 生成交叉表，然后绘制分组柱状图，直观展示每种疾病类型下不同治疗结果的数量分布。

7. **年龄与疾病严重程度散点图**：将疾病严重程度映射为数值（不严重=1，一般=2，严重=3，非常严重=4），绘制散点图分析两者关系。

8. **保存数据**：将日期转为字符串格式，删除辅助列，保存为CSV文件。
